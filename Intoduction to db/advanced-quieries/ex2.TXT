1. Find all students with GPA between 3.0 and 3.5
solutions: db.students.find({ gpa: { $gte: 3.0, $lte: 3.5 } })

2. Find students enrolled in more than 5 courses
solutions: db.students.find({ $expr: { $gt: [ { $size: "$courses" }, 5 ] } })

3. Get top 10 students by GPA
solutions: db.students.find().sort({ gpa: -1 }).limit(10)

4. Count students by city
solutions: db.students.aggregate([ { $group: { _id: "$city", totalStudents: { $sum: 1 } } } ])
